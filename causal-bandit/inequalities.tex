Assume we have i.i.d samples $X_i \in \{0,1\}$ from a Bernoulli distribution with mean $p$

Hoeffding's inequality
\eqn {
P(\frac{1}{n}\sum_{i=1}^n X_i -p \geq \epsilon) \leq e^{-2n\epsilon^2} 
}

Chebyshev
\eqn {
P(\frac{1}{n}\sum_{i=1}^n X_i -p \geq \epsilon) \leq \frac{1}{\epsilon}\sqrt{\frac{p(1-p)}{n}}
}

Chernoff 
\eqn {
P(\frac{1}{n}\sum_{i=1}^n X_i -p \geq \epsilon) \leq e^{-nD(p+\epsilon||p)} 
}
